components/WhoIWorkWith.tsx
import * as React from "react";
import { motion } from "framer-motion";
import { CheckCircle2, XCircle } from "lucide-react";

export interface WhoIWorkWithProps {
  variant?: "dark" | "light";
  compact?: boolean;
  className?: string;
}

const baseCopy = {
  with: [
    "Leaders who still feel the weight of responsibility — for families, teams, or nations.",
    "People who want strategy that honours both God and data, not one at the expense of the other.",
    "Boards and founders who can handle honest challenge, not curated encouragement.",
    "Fathers and builders who are more interested in consequence and legacy than in optics.",
  ],
  notWith: [
    "Anyone looking for hype, shortcuts, or purely cosmetic “brand” work.",
    "Leaders who only want validation, not diagnosis or accountability.",
    "Teams unwilling to confront hard realities — market, political, or personal.",
    "Situations where ethics, law, or basic integrity are negotiable tools.",
  ],
};

export default function WhoIWorkWith({
  variant = "dark",
  compact = false,
  className = "",
}: WhoIWorkWithProps): JSX.Element {
  const isDark = variant === "dark";

  const wrapperClasses = [
    "rounded-2xl border px-6 py-6 sm:px-8 sm:py-8 mt-10",
    isDark
      ? "border-gold/30 bg-charcoal/80 text-gold/80"
      : "border-deepCharcoal/10 bg-warmWhite text-deepCharcoal/80",
    className,
  ]
    .filter(Boolean)
    .join(" ");

  const headingClasses = [
    "font-serif font-semibold",
    compact ? "text-xl" : "text-2xl",
  ].join(" ");

  const subHeadingClasses = "text-sm font-semibold uppercase tracking-[0.18em] mb-3";

  const listClasses = "space-y-3 text-sm leading-relaxed";

  return (
    <motion.section
      className={wrapperClasses}
      initial={{ opacity: 0, y: 20 }}
      whileInView={{ opacity: 1, y: 0 }}
      transition={{ duration: 0.5 }}
      viewport={{ once: true }}
    >
      <div className="mb-6 max-w-3xl">
        <p className="text-xs font-semibold uppercase tracking-[0.22em] opacity-80">
          Fit matters
        </p>
        <h2 className={headingClasses}>
          Who I work with — and who I don&apos;t.
        </h2>
        {!compact && (
          <p className="mt-3 text-sm opacity-80">
            This is not for everyone. That&apos;s deliberate. Clarity protects
            both of us — and keeps the work honest, efficient, and effective.
          </p>
        )}
      </div>

      <div className="grid gap-8 md:grid-cols-2">
        <div>
          <h3 className={subHeadingClasses}>I usually work with</h3>
          <ul className={listClasses}>
            {baseCopy.with.map((item, index) => (
              <li key={index} className="flex items-start gap-3">
                <CheckCircle2
                  className={`mt-0.5 h-4 w-4 flex-shrink-0 ${
                    isDark ? "text-gold" : "text-forest"
                  }`}
                />
                <span>{item}</span>
              </li>
            ))}
          </ul>
        </div>

        <div>
          <h3 className={subHeadingClasses}>I&apos;m not the right fit for</h3>
          <ul className={listClasses}>
            {baseCopy.notWith.map((item, index) => (
              <li key={index} className="flex items-start gap-3">
                <XCircle
                  className={`mt-0.5 h-4 w-4 flex-shrink-0 ${
                    isDark ? "text-amber-300" : "text-amber-600"
                  }`}
                />
                <span>{item}</span>
              </li>
            ))}
          </ul>
        </div>
      </div>
    </motion.section>
  );
}
